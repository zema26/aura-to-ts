// Module 'Euclidean' is implied by the file context in TypeScript

class GCD {
    // data members
    a: number = 0;
    b: number = 0;

    // member function
    // Implements the Euclidean algorithm using subtraction
    euclid(a: number, b: number): number {
        
        while (b !== 0) {
            if (a > b) {
                a = a - b; // a - b -> a
            } else {
                b = b - a; // b - a -> b
            }
        }
        
        return a;
    }
}
