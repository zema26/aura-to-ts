Aura-to-TypeScript Code Translator

Aura is a conceptual programming language syntax designed to make code read more naturally, from left to right. It introduces a few key changes to common programming syntax, drawing inspiration from the flexibility of markup languages while maintaining the rigor of a traditional language. This explorer will walk you through its core concepts and visualize how they work in a real algorithm

This page demonstrates the translation of the custom Aura programming language into TypeScript. The Aura language syntax is defined by the provided files and includes several unique features:

    [cite_start]
  • Tagged Keywords: Control flow and definitions are wrapped in tags like {fun} ... {/fun} or {while} ... {/while}. Control flow keywords are enclosed in `{}` tags, similar to HTML or XML. This clearly defines the start and end of a block, making code structure explicit and easy to parse. [cite: 2]
    • [cite_start]
  • Reversed Assignment: Uses a -> operator, allowing assignments to flow from left to right (e.g., a + b -> a). This mimics how we might describe a calculation naturally: "take 'a minus b' and put it into 'a'". [cite: 2]
    • [cite_start]
  • Reversed Function Declaration: Syntax is {fun} <params> -> <name> <return_type>. Function declarations also follow the left-to-right pattern: parameters come first, followed by `->`, the function name, and its return type. This creates a consistent flow. [cite: 2]

Example 1: Euclidean Algorithm (from Aura1.txt)

[cite_start]

Aura Source Code [cite: 1]

{module} Euclidean 		// module itself

     {fun} int a, int b -> Euclid int  // function with type int
							
	{while} b!= 0
		{if} a > b
			a - b -> a
		{else}
			b - a -> b
		{/else}
		{/if}
	{/while}

	{return} a {/return}

    {/fun}

  
  {fun}	string args[] -> main int	// main function

	int a, b

        in -> a, b -> Euclid -> out

        // combined input and output and function call

	// here is introduced streamlined logic and
	
	// seamless input to output syntax

  {/fun}

{/module}

Translated TypeScript Code

// module Euclidean
export class Euclidean {

    // {fun} int a, int b -> Euclid int
    public Euclid(a: number, b: number): number {
        
        // {while} b!= 0
        while (b !== 0) {
            // {if} a > b
            if (a > b) {
                // a - b -> a
                a = a - b;
            } else {
                // b - a -> b
                b = b - a;
            }
        }
        // {/while}

        // {return} a {/return}
        return a;
    }

    // {fun} string args[] -> main int
    // Simulating the "in -> a, b -> Euclid -> out" pipeline
    public main(): number {
        // 'in' is simulated using prompts
        const a = parseInt(prompt("Enter value for a:") || "0");
        const b = parseInt(prompt("Enter value for b:") || "0");

        const result = this.Euclid(a, b);

        // 'out' is simulated using console.log
        console.log(result);
        return 0;
    }
}

// Simulation of running the main function
const euclidApp = new Euclidean();
euclidApp.main();

Translated TypeScript Code

// module primes
// Assuming a Node.js environment for 'process.stdout.write' 
// to match Aura's 'out' stream behavior (no auto-newline)
export class Primes {

    // {fun} int n -> Eratosthenes
    public Eratosthenes(n: number): void {
        
        // bool primes [ n ]
        let primes: boolean[] = new Array(n);
        
        // int l(0), i(0), index_square(3)
        let l: number = 0;
        let i: number = 0;
        let index_square: number = 3;

        // int first, last, factor
        let first: number;
        let last: number;
        let factor: number;

        // {for} k(0)++ < n
        for (let k = 0; k < n; k++) {
            // true -> primes[k]
            primes[k] = true;
        }
        // {/for}

        // {while} index_square < n
        while (index_square < n) {
            
            // {if} primes[i]
            if (primes[i]) {
                // 0 + index_square -> first
                first = 0 + index_square;
                // 0 + n -> last
                last = 0 + n;
                // i + i + 3 -> factor
                factor = i + i + 3;
                // false -> primes[first]
                primes[first] = false;

                // {while} last - first > factor
                while (last - first > factor) {
                    // first + factor -> first
                    first = first + factor;
                    // false -> primes[first]
                    primes[first] = false;
                }
            }

            // i + 1 -> i
            i = i + 1;
            // 2 * i * (i + 3) + 3 -> index_square
            index_square = 2 * i * (i + 3) + 3;
        } // {/while}
        
        // ' 2' -> out
        process.stdout.write(" 2");

        // {for} i(0)++ < n
        for (let i = 0; i < n; i++) {
            // {if} primes[i]
            if (primes[i]) {
                
                // {if} 2 * i + 3 > n
                if (2 * i + 3 > n) {
                    // break
                    break;
                }

                // ' ' 2 * i + 3 -> out
                process.stdout.write(" " + (2 * i + 3));
                // l + 1 -> l
                l = l + 1;

                // {if} l % 10 == 0
                if (l % 10 === 0) {
                    // '\n' -> out
                    process.stdout.write("\n");
                }
            }
        } // {/for}

        // '\n number : ' l -> out
        process.stdout.write("\n number : " + l);
    }

    // {fun} main int
    public main(): number {
        // 1000 -> Eratosthenes
        this.Eratosthenes(1000);
        return 0;
    }
}

// Simulation of running the main function
const primesApp = new Primes();
primesApp.main();
